class NavigationEffect {
  constructor(navigation) {
    this.previous = null;
    this.current = null;
    this.navigation = navigation;
    this.anchors = this.navigation.querySelectorAll("a");

    this.anchors.forEach((anchor) => {
      anchor.addEventListener("click", (e) => {
        this.handlePrevious();
        this.handleCurrent(anchor);
      });
    });
  }

  handleCurrent(current) {
    this.current = current;
    this.current.classList.toggle("active");
    const nodes = this.getNodes(this.current);

    gsap.to(nodes[0], {
      duration: 1.8,
      ease: "elastic.out(1.4, 0.4)",
      yPercent: "-100",
      stagger: 0.008,
      overwrite: true
    });

    gsap.to(nodes[1], {
      duration: 1.8,
      ease: "elastic.out(1.4, 0.4)",
      yPercent: "-100",
      stagger: 0.008,
      overwrite: true
    });
  }

  handlePrevious() {
    this.previous = document.querySelector(".active");
    if (this.previous) {
      this.previous.classList.toggle("active");
      const nodes = this.getNodes(this.previous);
      gsap.to(nodes[0], {
        duration: 0.2,
        ease: "power1.out",
        yPercent: "100",
        overwrite: true,
        stagger: 0.012
      });

      gsap.to(nodes[1], {
        duration: 0.2,
        ease: "power1.out",
        yPercent: "100",
        overwrite: true,
        delay: 0.02,
        stagger: 0.012
      });
    }
  }

  getNodes(item) {
    return [
      gsap.utils.shuffle(gsap.utils.selector(item)(".blue rect")),
      gsap.utils.shuffle(gsap.utils.selector(item)(".pink rect"))
    ];
  }
}

document.addEventListener("DOMContentLoaded", () => {
  gsap.registerPlugin(EasePack);
  new NavigationEffect(document.querySelector("nav"));
});



微信小程序支持自定义底部导航栏，开发者可通过配置和组件开发实现不同程度的个性化设计。以下是详细说明：
一、基础配置实现样式自定义
通过全局配置文件app.json中的tabBar字段，可设置导航栏的颜色、图标和文字等基础样式：
json
{
  "tabBar": {
    "custom": false, // 关闭自定义组件模式（默认）
    "color": "#7A7E83", // 未选中文字颜色
    "selectedColor": "#d81e06", // 选中文字颜色
    "backgroundColor": "#ffffff", // 背景色
    "borderStyle": "black", // 上边框颜色
    "list": [
      {
        "text": "首页",
        "pagePath": "pages/index/index",
        "iconPath": "/images/01.png",
        "selectedIconPath": "/images/01_select.png"
      }
    ]
  }
}
图标限制：单个图标文件不超过 40KB，建议尺寸 81px×81px。
Tab 数量：最多配置 5 个 Tab，最少 2 个。
二、深度自定义：使用自定义组件
若需动态切换导航栏内容（如多角色适配），可启用custom: true并创建独立组件：
目录结构：在项目根目录创建custom-tab-bar文件夹，包含index.wxml、index.js等组件文件。
组件逻辑：通过wx.getStorageSync获取用户角色，动态渲染不同 Tab 列表：
javascript
// custom-tab-bar/index.js
Component({
  attached() {
    const roleId = wx.getStorageSync('status');
    this.setData({
      list: roleId === 0 ? this.data.customerList : this.data.merchantList
    });
  }
});
样式优化：适配安全区域（如 iPhone X 底部留白）：
css
/* custom-tab-bar/index.wxss */
.tab-bar {
  padding-bottom: env(safe-area-inset-bottom);
}
三、进阶功能实现
1. 动态交互与路由控制
切换 Tab：通过wx.switchTab实现页面跳转，并更新导航栏选中状态：
javascript
// 组件方法
switchTab(e) {
  const path = e.currentTarget.dataset.path;
  this.setData({ selected: e.currentTarget.dataset.index });
  wx.switchTab({ url: path });
}
红点提示：使用wx.showTabBarRedDot在指定 Tab 上显示红点。
2. 多机型适配
高度差异：iOS 导航栏高度为 44px，Android 为 48px，需动态计算：
javascript
const sysInfo = wx.getSystemInfoSync();
const navHeight = sysInfo.system.includes('iOS') ? 44 : 48;
单位规范：使用px而非rpx确保与原生胶囊按钮尺寸一致。
3. 性能优化策略
懒加载：非首屏图标延迟加载，减少初始渲染耗时。
缓存机制：将常用图标缓存至本地，避免重复请求。
避免复杂动画：过度动画可能导致切换卡顿，建议采用基础渐变效果。
四、注意事项与限制
审核规范：
导航栏必须保留返回和关闭功能，禁止完全隐藏原生导航栏。
自定义样式需与微信整体风格协调，避免误导用户。
功能限制：
无法修改原生胶囊按钮（右上角菜单）的位置和样式。
跳转其他小程序需通过wx.navigateToMiniProgram，并在配置中声明白名单。
兼容性处理：
动态获取系统信息时，需处理wx.getSystemInfoSync的兼容性。
部分 Android 机型状态栏高度可能异常，建议设置最小高度保底。
五、典型应用场景
多角色适配：
用户版显示 “首页 / 校园 / 我的”，商家版显示 “订单 / 个人”。
通过wx.getStorageSync获取用户角色，动态切换 Tab 列表。
中间悬浮按钮：
实现类似微信 “发现” 页的中间凸起按钮，需在组件中调整布局：
wxml
<view class="tab-bar-item center-btn" bindtap="goToPublish">
  <image src="/images/publish.png"></image>
</view>
