### 三、交互逻辑增强
添加搜索框点击事件处理，以及动态计算布局：

```javascript
// 页面逻辑
Page({
  data: {
    pageTitle: "商城首页",
    statusBarHeight: 0,
    navBarHeight: 0,
    menuButtonInfo: {}
  },
  
  onLoad() {
    this.calculateLayout();
  },
  
  // 计算导航栏布局信息
  calculateLayout() {
    const systemInfo = wx.getSystemInfoSync();
    const menuButtonInfo = wx.getMenuButtonBoundingClientRect();
    
    // 计算导航栏高度
    const navBarHeight = menuButtonInfo.height + (menuButtonInfo.top - systemInfo.statusBarHeight) * 2;
    
    this.setData({
      statusBarHeight: systemInfo.statusBarHeight,
      navBarHeight: navBarHeight,
      menuButtonInfo: menuButtonInfo
    });
  },
  
  // 打开搜索页面
  openSearch() {
    wx.navigateTo({
      url: '/pkgContent/search/search'
    });
  }
});
```

### 五、搜索框扩展功能
可扩展搜索框为实时搜索组件，添加以下代码：

```wxml
<!-- 带输入功能的搜索框 -->
<view class="search-box">
  <text class="search-icon">🔍</text>
  <input 
    type="text" 
    placeholder="搜索商品" 
    bindinput="onSearchInput" 
    bindconfirm="onSearchConfirm"
    confirm-type="search"
  />
  <view class="clear-btn" wx:if="{{searchValue.length > 0}}" bindtap="clearSearch">
    <text>×</text>
  </view>
</view>
```

```javascript
// 搜索框事件处理
Page({
  data: {
    searchValue: ''
  },
  
  // 搜索输入事件
  onSearchInput(e) {
    this.setData({
      searchValue: e.detail.value
    });
  },
  
  // 清空搜索内容
  clearSearch() {
    this.setData({
      searchValue: ''
    });
  },
  
  // 确认搜索
  onSearchConfirm() {
    if (this.data.searchValue.trim()) {
      // 执行搜索逻辑
      this.performSearch(this.data.searchValue);
    }
  },
  
  // 执行搜索
  performSearch(keyword) {
    // 调用搜索API或跳转搜索结果页
    console.log('搜索:', keyword);
    wx.navigateTo({
      url: `/pages/search/result?keyword=${encodeURIComponent(keyword)}`
    });
  }
});
```
