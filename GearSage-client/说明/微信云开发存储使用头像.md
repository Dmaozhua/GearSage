##### 步骤 2：上传到云存储
```javascript
// 上传方法
async uploadToCloud(tempFilePath) {
  try {
    // 1. 生成唯一文件名
    const timestamp = Date.now()
    const random = Math.floor(Math.random() * 1000)
    const cloudPath = `avatar/${timestamp}-${random}.jpg`
    
    // 2. 执行上传
    const uploadRes = await wx.cloud.uploadFile({
      cloudPath,          // 云端路径
      filePath: tempFilePath, // 本地临时路径
    })
    
    // 3. 获取永久文件ID (格式: cloud://env-id/file-path)
    const fileID = uploadRes.fileID
    console.log('上传成功', fileID)
    
    // 4. 更新用户信息
    this.updateUserAvatar(fileID)
  } catch (err) {
    console.error('上传失败', err)
    wx.showToast({ title: '头像上传失败', icon: 'error' })
  }
}
```

##### 步骤 3：存储到数据库
```javascript
// 更新用户信息
async updateUserAvatar(fileID) {
  try {
    const db = wx.cloud.database()
    const _ = db.command
    
    // 获取当前用户openid
    const { openid } = await this.getUserOpenId()
    
    // 更新或创建用户记录
    await db.collection('users').where({
      _openid: openid
    }).update({
      data: {
        avatar: fileID,  // 存储云文件ID
        lastUpdate: _.set(new Date())
      }
    })
    
    wx.showToast({ title: '头像更新成功' })
  } catch (err) {
    console.error('数据库更新失败', err)
  }
}

// 获取用户openid
async getUserOpenId() {
  const res = await wx.cloud.callFunction({
    name: 'login'
  })
  return res.result
}
```

  })
  return res.fileList[0].tempFileURL
}

2. **错误重试机制**
   ```javascript
   async uploadWithRetry(filePath, retries = 2) {
     try {
       return await wx.cloud.uploadFile({...})
     } catch (err) {
       if (retries > 0) {
         console.log(`重试上传 (剩余${retries}次)`)
         return this.uploadWithRetry(filePath, retries - 1)
       }
       throw err
     }
   }
   ```
